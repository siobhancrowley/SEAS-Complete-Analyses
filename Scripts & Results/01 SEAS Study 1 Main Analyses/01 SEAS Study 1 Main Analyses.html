To assess other factors that may potentially affect participants’ performance on memory check questions, we created a generalized linear mixed model (GLMM) with a binomial family, using trial, type (cyber, animals, playdoh, and trampoline), siblings, gender, and age in months as predictors. This model failed to converge, so all variables except trial and age_months were removed; this also did not converge, so we tried to substitue a categorical age variable (3-6) in the same model, but this still did not converge. Therefore, we ran two different GLMMs, one for trial and one for age. In the trial model, only trial 3 (i.e. the first story) was significant (z = -2.38, p = 0.017). The GLMM examining age in months was not significant.

  mutate(
    resource_1 = if_else(resource_1 == "excluder", 1, 0, NA), #One participant who did not answer the first resource question
    resource_2 = if_else(resource_2 == "excluder", 1, 0),
    resource_3 = if_else(resource_3 == "excluder", 1, 0),
    resource_4 = if_else(resource_4 == "excluder", 1, 0),
    resource_total = resource_1 + resource_2 + resource_3 + resource_4
  )

#Physical and social resources sum scores
seas <- seas |>
  mutate(
    resource_physical = case_when(
      resource_type_1 == "physical" ~ resource_1,
      resource_type_2 == "physical" ~ resource_2
    ),
    resource_social = case_when(
      resource_type_1 == "social" ~ resource_1,
      resource_type_2 == "social" ~ resource_2
    )
  ) |>
  mutate(
    resource_physical = case_when(
      resource_type_3 == "physical" ~ resource_physical + resource_3,
      resource_type_4 == "physical" ~ resource_physical + resource_4
    ),
    resource_social = case_when(
      resource_type_3 == "social" ~ resource_social + resource_3,
      resource_type_4 == "social" ~ resource_social + resource_4
    )
  )
