Data Preparation

seas2_full <- read_excel("../Data/FINAL SEAS V2 Dataset.xlsx")

seas2 <- subset(seas2_full, final_data == "Y") # dropping excluded participants
#EmQue Scores
seas2 <- seas2 |>
  mutate(across(starts_with("EQ_"), ~case_when(
    . == "Never" ~ 0,
    . == "Sometimes" ~ 1,
    . == "Often" ~ 2
  ))) |> #convert EmQue columns to numeric
  rowwise() |>
  mutate(
      EQ_total = sum(c_across(starts_with("EQ_"))),
      EQ_contagion = sum(c_across(c(EQ_1, EQ_4, EQ_7, EQ_10, EQ_13, EQ_16, EQ_19))),
      EQ_attention = sum(c_across(c(EQ_3, EQ_6, EQ_9, EQ_12, EQ_15, EQ_18, EQ_20))), 
      EQ_prosocial = sum(c_across(c(EQ_2, EQ_5, EQ_8, EQ_11, EQ_14, EQ_17))),
      .after = EQ_20
    ) |>
  ungroup()

# Memory Check
seas2 <- seas2 |>
  mutate(across(
    .cols = c(starts_with("mem_check"), starts_with("comp_mem_check")),
    .fns = ~if_else(.x == "correct", 0.5, 0)
         )) |> #convert to numeric
  mutate(
    mem_check_1 = mem_check_1A + mem_check_1B,
    mem_check_2 = mem_check_2A + mem_check_2B,
    comp_mem_check_1 = comp_mem_check_1A + comp_mem_check_1B,
    mem_check_3 = mem_check_3A + mem_check_3B,
    mem_check_4 = mem_check_4A + mem_check_4B,
    comp_mem_check_2 = comp_mem_check_2A + comp_mem_check_2B,
    mem_check_total = mem_check_1 + mem_check_2 + mem_check_3 + mem_check_4 + comp_mem_check_1 + comp_mem_check_2
  ) #sum scores

# Status
## Exclusion Trials
seas2 <- seas2 |>
  mutate(
    exclusion_video_selection = case_when(
      interaction_1 == "exclusion" ~ status_1, 
      interaction_2 == "exclusion" ~ status_2
    ),
    exclusion_story_selection = case_when(
      interaction_3 == "exclusion" ~ status_3, 
      interaction_4 == "exclusion" ~ status_4
    )
  ) |> #select exclusion from videos and stories
  mutate(across(
    .cols = c(exclusion_video_selection, exclusion_story_selection),
    .fns = ~if_else(.x == "excluder", 1, 0)
  )) |> #convert numeric
  rowwise() |>
  mutate(
    exclusion_status_total = sum(c_across(c(exclusion_video_selection, exclusion_story_selection)))
  ) |> #sum
  ungroup()

## Comparison Trials
seas2 <- seas2 |>
  mutate(
    comp_status_1_num = if_else(comp_status_1 == "excluder", 1, 0),
    comp_status_2_num = if_else(comp_status_2 == "excluder", 1, 0),
    comp_status_total = comp_status_1_num + comp_status_2_num
  )

# Resources
## Exclusion Trials
seas2 <- seas2 |>
  mutate(
    exclusion_physical_selection = case_when(
      interaction_1 == "exclusion" & resource_type_1 == "physical" ~ resource_1, 
      interaction_2 == "exclusion" & resource_type_2 == "physical" ~ resource_2,
      interaction_3 == "exclusion" & resource_type_3 == "physical" ~ resource_3,
      interaction_4 == "exclusion" & resource_type_4 == "physical" ~ resource_4
    ),
    exclusion_social_selection = case_when(
      interaction_1 == "exclusion" & resource_type_1 == "social" ~ resource_1, 
      interaction_2 == "exclusion" & resource_type_2 == "social" ~ resource_2,
      interaction_3 == "exclusion" & resource_type_3 == "social" ~ resource_3,
      interaction_4 == "exclusion" & resource_type_4 == "social" ~ resource_4
    )
  ) |> #pull out physical and social excluder selection responses
  mutate(across(
    .cols = c(exclusion_physical_selection, exclusion_social_selection),
    .fns = ~if_else(.x == "excluder", 1, 0)
  )) |> #convert to numeric
  rowwise() |>
  mutate(
    exclusion_resource_total = exclusion_physical_selection + exclusion_social_selection
  ) |>
  ungroup() #create total resources score

## Comparison Trials
seas2 <- seas2 |>
  mutate(
    comp_physical_selection = if_else(comp_resource_type_1 == "physical", comp_resource_1, comp_resource_2),
    comp_social_selection = if_else(comp_resource_type_1 == "social", comp_resource_1, comp_resource_2)
  ) |>
  mutate(
    across(
      .cols = c(comp_physical_selection, comp_social_selection), 
      .fns = ~if_else(.x == "excluder", 1, 0)
    ),
    comp_resource_total = comp_physical_selection + comp_social_selection
  )

Empathy Correlations

EmQue Total

labels = c("Memory Check", "Exclusion Status Selection", "Comparison Status Selection", "Exclusion Resource Selection", "Exclusion Physical Selection", "Exclusion Social Selection", "Comparison Resource Selection", "Comparison Physical Selection", "Comparison Social Selection")

total_outcomes <- seas2 |>
  select(EQ_total, mem_check_total, exclusion_status_total, comp_status_total, exclusion_resource_total, exclusion_physical_selection, exclusion_social_selection, comp_resource_total, comp_physical_selection, comp_social_selection)

total_cor_matrix <- rcorr(as.matrix(total_outcomes), type = "pearson")

total_correlation <- total_cor_matrix$r[1, -1]
total_p <- total_cor_matrix$P[1, -1]
total_degrees <- total_cor_matrix$n[1, 1] - 2 #I did confirm that all the ns were the same for the values within each of the 4 matrices

total_df <- data.frame(r = total_correlation, p = total_p) 

total_df <- total_df |>
  rownames_to_column(var = "variable") |>
  mutate(
    variable = labels,
    r = signif(r, 2),
    p = signif(p, 2),
    "Corrected p" = if_else(p * 9 > 1, 1.0, p * 9)
  )

total_cor_table <- total_df |>
  gt(
    rowname_col = "variable",
  )|>
    opt_stylize(style = 6, color = "blue") |>
    tab_header(title = "EmQue Total Correlations") |>
  tab_style(
    style = cell_text(weight = "bold"),
    locations = cells_body(
      columns = c(r, p, "Corrected p"),
      rows = p < 0.05
    )
  ) |>
  cols_align(
    align = "center",
    columns = everything()
  ) |> 
  tab_footnote(
    "Correlations were calculated with the sum scores of each respective variable"
  ) |>
  tab_footnote(
    "Corrections used the Bonferroni method, k = 9"
  ) |>
  tab_footnote(
    str_glue("Degrees of freedom = {total_degrees}")
  )

EmQue total scores correlated with two variables: comparison resource selection total (r = -0.23 (weak negative), p = 0.03, p with Bonferroni correction = 0.27) and comparison resource physical selection (r = -0.27 (weak negative), p = 0.012, p with Bonferroni correction = 0.11), such that participants with higher empathy scores were less likely to select the excluder over the includer as having more resources overall and more physical resources.
